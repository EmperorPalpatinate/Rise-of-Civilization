country_event = {
	id = roc_religion.1
	title = roc_religion.1.t
	desc = roc_religion.1.d
	picture = FORT_eventPicture
	fire_only_once = no
	
	trigger = {
		NOT = { is_year = 500 }
		NOT = {
			has_country_flag = new_religion
		}
		NOT = {
			any_country = {
				religion = ashurism
			}
			any_province = {
				religion = ashurism
			}
		}
	}
	
	mean_time_to_happen = {
		months = 5000
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = new_religion
			change_religion = ashurism
		}
	}
	option = {
		name = roc_religion.1.a		
		ai_chance = {			
			factor = 1
		}
		random_owned_province = {
			change_religion = ashurism
		}
	}
}
