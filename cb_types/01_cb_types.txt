# Foreign Invasion
cb_foreign_invasion = {
	valid_for_subject = {
	
	prerequisites = {
		FROM = {
			num_of_owned_provinces_with = {
				value = 75
				NOT = { culture = root }
				is_overseas = no
			}
		}
		is_neighbor_of = FROM
		ROOT = {
			NOT = {
				num_of_owned_provinces_with = {
					value = 75
				}
			}
		}	
	}
	
	war_goal = foreign_invasion
}
		
# Revolt War
cb_revolt_war = {
	valid_for_subject = no
	exclusive = yes

	prerequisites = {
		ROOT = {
			has_country_flag = fote_civil_war
			NOT = { 
				had_country_flag = { 
					flag = fote_civil_war
					days = 14600
				}
			}
		}
		FROM = {
			has_country_flag = fote_civil_war
			NOT = { 
				had_country_flag = { 
					flag = fote_civil_war
					days = 14600
				}
			}
		}
		OR = {
			AND = {
				ROOT = {
					has_country_flag = fote_crumbling_empire
					NOT = {
						had_country_flag = {
							flag = fote_crumbling_empire
							days = 14600
						}
					}
				}
				FROM = {
					OR = {
						NOT = {
							has_country_flag = fote_crumbling_empire
						}
						had_country_flag = {
							flag = fote_crumbling_empire
							days = 14600
						}
					}
				}
			}
			AND = {
				ROOT = {
					OR = {
						NOT = {
							has_country_flag = fote_crumbling_empire
						}
						had_country_flag = {
							flag = fote_crumbling_empire
							days = 14600
						}
					}
				}
				FROM = {
					has_country_flag = fote_crumbling_empire
					NOT = {
						had_country_flag = {
							flag = fote_crumbling_empire
							days = 14600
						}
					}
				}
			}
			AND = {
				ROOT = {
					NOT = {
						has_country_flag = fote_crumbling_empire
					}
				}
				FROM = {
					NOT = {
						has_country_flag = fote_crumbling_empire
					}
				}
			}
		}
	}


	can_use_peace_treaty = {
		who = FROM
		treaty = revolt_annex
		treaty = occupied_annex
	}

	attacker_disabled_po = {
		po_release_vassals
		po_release_annexed
		po_become_vassal
		po_subjugate_vassal
		po_become_tributary_state
		po_subjugate_tributary_state
		po_form_personal_union
	}
	war_goal = defend_capital_revolt
}

# Imperial Rome cb
cb_roman_imperial = {
	valid_for_subject = no
	
	prerequisites_self = {
		has_country_flag = rome_formed
		tag = ROM
		is_subject = no
		is_revolutionary = no
	}
	
	prerequisites = {
		FROM = { #Imperial borders
			NOT = { is_subject_of = ROOT }
			any_owned_province = {
				OR = {
					region = france_region
					region = iberia_region
					region = italy_region
					region = balkan_region
					region = low_countries_region
					region = british_isles_region
					region = anatolia_region
					region = mashriq_region
					region = egypt_region
					region = maghreb_region
					region = north_german_region
					region = south_german_region
				}
			}
		}
		is_neighbor_of = FROM
	}
	
	war_goal = superiority_roman_imperialism
}
